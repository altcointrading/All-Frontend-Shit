var sitetitle      = document.getElementById("title"),
    stop      = sitetitle.offsetTop - 60,
    docBody   = document.documentElement || document.body.parentNode || document.body,
    hasOffset = window.pageYOffset !== undefined,
    scrollTop;



window.onscroll = function (e) {
  scrollTop = hasOffset ? window.pageYOffset : docBody.scrollTop;

  console.log(scrollTop);
  if (scrollTop >= stop) {
    sitetitle.className = 'stick';
  } else {
    sitetitle.className = 'site-title';
  }
}

// enqueue the javascript in wp footer

function js_logo_slide() {
	wp_enqueue_script(
		'LogoSlideDown',
		get_template_directory_uri().'/js/logoslidedown.js',
		false,
        '1.0',
        true
	);
}
add_action( 'wp_enqueue_scripts', 'js_logo_slide' );
